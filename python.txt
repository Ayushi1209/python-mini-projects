---
- name: Fetch Resource Group of ASG
  hosts: localhost
  gather_facts: false
  tasks:
    - name: Get Resource Group
      command: "az network asg show --name {{ your_asg_name }} --query 'resourceGroup' -o tsv"
      register: resource_group_name
      failed_when: resource_group_name.rc != 0

    - name: Display Resource Group Name
      debug:
        var: resource_group_name.stdout
