- name: Retrieve Resource ID
  command: az network asg show -n {{ asg_name }} -g {{ resource_group }} --query 'id'
  register: result1

- name: Clean the Resource ID
  set_fact:
    resource_id: "{{ result1.stdout_lines[0] | regex_replace('^\"|\"$', '') }}"

- name: Print resource id
  debug:
    var: resource_id
