---
- name: Fetch Resource Group of ASG
  hosts: localhost
  gather_facts: false
  tasks:
    - name: Get ASG Facts
      azure_rm_securitygroup_facts:
        name: "{{ your_asg_name }}"
      register: asg_facts
      failed_when: asg_facts.securitygroups | length == 0

    - name: Display Resource Group Name
      debug:
        var: asg_facts.securitygroups[0].resource_group
