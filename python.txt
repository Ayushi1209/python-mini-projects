---
- name: Get Azure ASG Resource ID
  hosts: localhost
  tasks:
    - name: Get Azure ASG Resource ID
      azure_rm_securitygroup_info:
        resource_group: "{{ resource_group_name }}"
        name: "{{ asg_name }}"
      register: asg_info

    - name: Display ASG Resource ID
      debug:
        var: asg_info.ansible_facts.id
